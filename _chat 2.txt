    sorted_arr = np.sort(arr)[::-1]

    # Compute the total sum of the input array
    total_sum = np.sum(sorted_arr)

    # Initialize a 2D array to keep track of the minimum difference
    # between the sum of two subsets for all possible subset sums
    dp = np.full((total_sum + 1, len(sorted_arr) + 1), np.inf)
    dp[0, 0] = 0

    # Iterate through the input array and fill in the 2D array
    for i in range(1, len(sorted_arr) + 1):
        for j in range(total_sum + 1):
            if sorted_arr[i - 1] > j:
                dp[j, i] = dp[j, i - 1]
            else:
                dp[j, i] = min(dp[j, i - 1], dp[j - sorted_arr[i - 1], i - 1] + 1)

    # Find the index of the minimum difference between two subsets
    # (which corresponds to the closest possible sum to half the total sum)
    mid = total_sum // 2
    while dp[mid, len(sorted_arr)] == np.inf:
        mid -= 1

    # Initialize variables to keep track of the two subsets
    subset1 = []
    subset2 = []

    # Trace back through the 2D array to find the two subsets
    i = len(sorted_arr)
    j = mid
    while i > 0 and j > 0:
        if dp[j, i] == dp[j, i - 1]:
            i -= 1
        else:
            subset1.append(sorted_arr[i - 1])
            j -= sorted_arr[i - 1]
            i -= 1
    
    # The remaining elements in the input array form the second subset
    subset2 = np.setdiff1d(sorted_arr, subset1)

    # Return the two subsets as NumPy arrays
    return np.array(subset1), subset2
